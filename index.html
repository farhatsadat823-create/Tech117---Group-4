Introduction

People with visual impairments often face significant challenges in communication. To interact with others, they rely on a tactile reading and writing system called Braille to spell out words. However, not everyone is able to read Braille. This creates a barrier between individuals with disabilities and the general population.

The proposed solution is a button-based input device connected to a display, which shows each letter as it is entered through the buttons, forming words.

The prototype is build using an Arduino Uno, black/blue/green/red/yellow pushbuttons, an LCD display screen, and a passive buzzer.

Circuit Diagram

The LCD displays the corresponding letters and numbers, and the Arduino processes the button inputs to trigger the appropriate buzzer signals.

Sustainability Assessment

  • Materials: The Arduino Uno is RoHS and WEEE compliant, limiting hazardous substances.
  • Energy Efficiency: The circuit draws minimal power (~50 mA at 5V), ideal for USB or solar power.
  • Reusability: All components are modular and reusable, reducing electronic waste.
  • Packaging & Shipping: Ordering all components from a single supplier helps minimize carbon footprint.

Arduino Code

The following Arduino code controls the system, lighting LEDs and activating the buzzer based on distance readings from the HC-SR04 sensor.

  
  
#include 

LiquidCrystal lcd(7, 6, 5, 4, 3, 2);

/* Braille buttons mapped to standard Braille dots:
 1 4
 2 5
 3 6 */

const int buttonPins[6] = {13, 12, 11, 10, 9, 8}; // dots 1–6
int prevButtonState[6] = {0, 0, 0, 0, 0, 0};

const int redButton    = A4; //Enter/confirm
const int greenButton  = A3; //Space
const int blackButton  = A2; //Toggle number/letter mode
const int squareButton = A0; //Delete

const int buzzer = A5;

//--------
// BUFFERS: temporary store all the characters typed so far
//--------

String currentWord = "";
bool numberMode = false; // false = letter mode, true = number mode

struct BrailleChar {
  byte pattern;
  char letter;
  char number;
};

//---------------
//BRAILLE MAPPING
//---------------
BrailleChar brailleMap[] = {
  {B000001, 'a', '1'},
  {B000011, 'b', '2'},
  {B000101, 'c', '3'},
  {B000111, 'd', '4'},
  {B000110, 'e', '5'},
  {B001101, 'f', '6'},
  {B001111, 'g', '7'},
  {B001110, 'h', '8'},
  {B000010, 'i', '9'},
  {B000110, 'j', '0'},
  {B001001, 'k', '-'},
  {B001011, 'l', '-'},
  {B001101, 'm', '-'},
  {B001111, 'n', '-'},
  {B001110, 'o', '-'},
  {B011001, 'p', '-'},
  {B011011, 'q', '-'},
  {B011101, 'r', '-'},
  {B011100, 's', '-'},
  {B010111, 't', '-'},
  {B111001, 'u', '-'},
  {B111011, 'v', '-'},
  {B101111, 'w', '-'},
  {B111101, 'x', '-'},
  {B111111, 'y', '-'},
  {B111110, 'z', '-'}
};

int mapSize = sizeof(brailleMap) / sizeof(brailleMap[0]);

//-----
//SOUND
//-----
void beep(int duration){
  tone(buzzer, 1000, duration);
  delay(duration + 20);
}

void doubleBeep(int d1, int d2){
  beep(d1);
  delay(100);
  beep(d2);
}

void longBeep(){
  tone(buzzer, 900, 400);
  delay(450);
}

void doubleLongBeep(){
  longBeep();
  delay(150);
  longBeep();
}

//---------------
//BRAILLE READING
//---------------
byte readBraillePattern() {
  byte pattern = 0;

  for (int i = 0; i < 6; i++) {
    int val = digitalRead(buttonPins[i]);

    if (val == HIGH)
      pattern |= (1 << i);

    if (val == HIGH && prevButtonState[i] == LOW) {
      beep(80);
      delay(30);
    }
    prevButtonState[i] = val;
  }
  return pattern;
}
 
//----------------
//PATTERN MATCHING
//----------------
char findCharacter(byte pattern) {
  for (int i = 0; i < mapSize; i++) {
    if (brailleMap[i].pattern == pattern) {
      return numberMode ? brailleMap[i].number : brailleMap[i].letter;
    }
  }
  return '?';
}

//-----
//SETUP
//-----
void setup() {
  // Braille buttons
  for (int i = 0; i < 6; i++)
    pinMode(buttonPins[i], INPUT);

  pinMode(redButton, INPUT);
  pinMode(greenButton, INPUT);
  pinMode(blackButton, INPUT);
  pinMode(squareButton, INPUT);

  pinMode(buzzer, OUTPUT);

  lcd.begin(16, 2);
  lcd.clear();
  lcd.print("Braille Input");
  delay(800);
  lcd.clear();
}

void loop() {
  byte pattern = readBraillePattern();

  // Red button = finish current letter
  if (digitalRead(redButton) == HIGH) {
    if (pattern > 0) {
      char c = findCharacter(pattern);
      currentWord += c;

      lcd.clear();
      lcd.print(currentWord);

      longBeep(); // letter confirm
    }

    while (digitalRead(redButton) == HIGH);
    delay(200);
  }

  // Green button = space
  if (digitalRead(greenButton) == HIGH) {
    currentWord += ' ';

    lcd.clear();
    lcd.print(currentWord);

    beep(100);

    while (digitalRead(greenButton) == HIGH);
    delay(200);
  }
  
  // Black button = toggle between letter & number mode
  if (digitalRead(blackButton) == HIGH) {
    numberMode = !numberMode;

    lcd.clear();

    if (numberMode) {
      lcd.print("Numbers Mode");
      doubleBeep(120, 120); // 2 short
    } else {
      lcd.print("Letters Mode");
      longBeep(); // 1 long
    }

    while (digitalRead(blackButton) == HIGH);
    delay(300);

    lcd.clear();
    lcd.print(currentWord);
  }

  // Red Square =  delete
  if (digitalRead(squareButton) == HIGH) {
    delay(250);

    // Long hold = delete whole word
    if (digitalRead(squareButton) == HIGH) {
      currentWord = "";
      lcd.clear();
      doubleLongBeep();
    }

    // Short press = delete last character
    else {
      if (currentWord.length() > 0) {
        currentWord.remove(currentWord.length() - 1);
      }

      lcd.clear();
      lcd.print(currentWord);

      doubleBeep(100, 100);
    }

    while (digitalRead(squareButton) == HIGH);
    delay(200);
  }
}

Conclusion

The system effectively demonstrates distance-based alerting using Arduino. It’s affordable, educational, and sustainable through reusable components.
