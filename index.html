Introduction

People with visual impairments often face significant challenges in communication. To interact with others, they rely on a tactile reading and writing system called Braille to spell out words. However, not everyone is able to read Braille. This creates a barrier between individuals with disabilities and the general population.

The proposed solution is a button-based input device connected to a display, which shows each letter as it is entered through the buttons, forming words.

The prototype is build using an Arduino Uno, black/blue/green/red/yellow pushbuttons, an LCD display screen, and a passive buzzer.

Circuit Diagram

The LCD displays the corresponding letters and numbers, and the Arduino processes the button inputs to trigger the appropriate buzzer signals.

Sustainability Assessment

  • Materials: The Arduino Uno is RoHS and WEEE compliant, limiting hazardous substances.
  • Energy Efficiency: The circuit draws minimal power (~50 mA at 5V), ideal for USB or solar power.
  • Reusability: All components are modular and reusable, reducing electronic waste.
  • Packaging & Shipping: Ordering all components from a single supplier helps minimize carbon footprint.

Arduino Code

The following Arduino code controls the system, lighting LEDs and activating the buzzer based on distance readings from the HC-SR04 sensor.

  
  

#include 

LiquidCrystal lcd(7, 6, 5, 4, 3, 2);

/* Braille dots layout:
 1 4
 2 5
 3 6
*/

// Button pins in dot order
const int buttonPins[6] = {13, 12, 11, 10, 9, 8}; 
int prevButtonState[6] = {0, 0, 0, 0, 0, 0};

// Function buttons
const int redButton    = A4; // confirm letter
const int greenButton  = A3; // space
const int blackButton  = A2; // switch letter/number
const int squareButton = A0; // delete

const int buzzer = A5;

String currentWord = "";
bool numberMode = false;

struct BrailleChar {
  byte pattern; // 6-bit braille pattern
  char letter;
  char number;
};

/*
 Bit order used:
 dot1 = bit0 = 1
 dot2 = bit1 = 2
 dot3 = bit2 = 4
 dot4 = bit3 = 8
 dot5 = bit4 = 16
 dot6 = bit5 = 32
*/

// CORRECT BRAILLE TABLE (Grade 1, A–Z)
BrailleChar brailleMap[] = {
  {B000001, 'a', '1'},  // 1
  {B000011, 'b', '2'},  // 1 2
  {B000101, 'c', '3'},  // 1 4
  {B000111, 'd', '4'},  // 1 4 5
  {B000110, 'e', '5'},  // 1 5
  {B001101, 'f', '6'},  // 1 2 4
  {B001111, 'g', '7'},  // 1 2 4 5
  {B001110, 'h', '8'},  // 1 2 5
  {B000011, 'i', '9'},  // 2 4
  {B000111, 'j', '0'},  // 2 4 5
  {B001001, 'k', '-'},  // 1 3
  {B001011, 'l', '-'},  // 1 2 3
  {B001101, 'm', '-'},  // 1 3 4
  {B001111, 'n', '-'},  // 1 3 4 5
  {B001110, 'o', '-'},  // 1 3 5
  {B011001, 'p', '-'},  // 1 2 3 4
  {B011011, 'q', '-'},  // 1 2 3 4 5
  {B011101, 'r', '-'},  // 1 2 3 5
  {B011100, 's', '-'},  // 2 3 4
  {B010111, 't', '-'},  // 2 3 4 5
  {B111001, 'u', '-'},  // 1 3 6
  {B111011, 'v', '-'},  // 1 2 3 6
  {B101111, 'w', '-'},  // 2 4 5 6
  {B111101, 'x', '-'},  // 1 3 4 6
  {B111111, 'y', '-'},  // 1 3 4 5 6
  {B111110, 'z', '-'}   // 1 3 5 6
};

int mapSize = sizeof(brailleMap) / sizeof(brailleMap[0]);

// ----------------------
// BEEPS
// ----------------------
void beep(int dur) {
  tone(buzzer, 1000, dur);
  delay(dur + 20);
}

void longBeep() {
  tone(buzzer, 900, 450);
  delay(500);
}

void doubleBeep() {
  beep(120);
  delay(100);
  beep(120);
}

void doubleLongBeep() {
  longBeep();
  delay(150);
  longBeep();
}

// ----------------------
// READ BRAILLE INPUT
// ----------------------
byte readBraillePattern() {
  byte pattern = 0;

  for (int i = 0; i < 6; i++) {
    int val = digitalRead(buttonPins[i]);

    if (val == LOW) {          // using INPUT_PULLUP
      pattern |= (1 << i);

      if (prevButtonState[i] == HIGH) {
        beep(80);              // button press detected
      }
    }

    prevButtonState[i] = val;
  }
  return pattern;
}

// ----------------------
// FIND CHARACTER
// ----------------------
char findCharacter(byte pattern) {
  for (int i = 0; i < mapSize; i++) {
    if (brailleMap[i].pattern == pattern) {
      return numberMode ? brailleMap[i].number : brailleMap[i].letter;
    }
  }
  return '?'; // unknown
}

// ----------------------
// SETUP
// ----------------------
void setup() {
  // Braille buttons
  for (int i = 0; i < 6; i++)
    pinMode(buttonPins[i], INPUT_PULLUP);

  pinMode(redButton,    INPUT_PULLUP);
  pinMode(greenButton,  INPUT_PULLUP);
  pinMode(blackButton,  INPUT_PULLUP);
  pinMode(squareButton, INPUT_PULLUP);

  pinMode(buzzer, OUTPUT);

  lcd.begin(16, 2);
  lcd.print("Braille Ready");
  delay(900);
  lcd.clear();
}

// ----------------------
// MAIN LOOP
// ----------------------
void loop() {
  byte pattern = readBraillePattern();

  // --- RED: confirm a letter ---
  if (digitalRead(redButton) == LOW) {
    if (pattern > 0) {
      char c = findCharacter(pattern);
      currentWord += c;

      lcd.clear();
      lcd.print(currentWord);
      longBeep(); // letter confirm
    }

    while (digitalRead(redButton) == LOW);
    delay(150);
  }

  // --- GREEN: space ---
  if (digitalRead(greenButton) == LOW) {
    currentWord += ' ';

    lcd.clear();
    lcd.print(currentWord);
    beep(100);

    while (digitalRead(greenButton) == LOW);
    delay(150);
  }

  // --- BLACK: toggle number/letter mode ---
  if (digitalRead(blackButton) == LOW) {
    numberMode = !numberMode;

    lcd.clear();
    if (numberMode) {
      lcd.print("Numbers Mode");
      doubleBeep();
    } else {
      lcd.print("Letters Mode");
      longBeep();
    }

    while (digitalRead(blackButton) == LOW);
    delay(250);

    lcd.clear();
    lcd.print(currentWord);
  }

  // --- SQUARE: delete ---
  if (digitalRead(squareButton) == LOW) {
    delay(250);

    // long hold = delete whole word
    if (digitalRead(squareButton) == LOW) {
      currentWord = "";
      lcd.clear();
      doubleLongBeep();
    } 
    else {
      // short press = delete last letter
      if (currentWord.length() > 0)
        currentWord.remove(currentWord.length() - 1);

      lcd.clear();
      lcd.print(currentWord);
      doubleBeep();
    }

    while (digitalRead(squareButton) == LOW);
    delay(150);
  }
}

Conclusion

The system effectively demonstrates distance-based alerting using Arduino. It’s affordable, educational, and sustainable through reusable components.
